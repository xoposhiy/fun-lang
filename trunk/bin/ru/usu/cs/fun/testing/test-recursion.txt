	if (<(n 2) 
		1
		+(fib(-(n 1)) fib(-(n 2)))
	).

		1 
		*(fact(-(n 1)) n)

fun gcd(a b)
	if(&(=(a 0) =(b 0)) 
		1
		if(=(a 0) 
			b
			if(>(a b) 
				gcd(b a) 
				gcd(%(b a) a)
			)
		)
	).
			
		
=(gcd(6 14) 2).
=(gcd(14 6) 2).
=(gcd(0 0) 1).
=(gcd(1 2) 1).
=(gcd(2 2) 2).
